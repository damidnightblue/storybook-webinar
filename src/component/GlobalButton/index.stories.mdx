export const Template = (args, { globals: { locale } }) => {
  const caption = getCaptionForLocale(locale);
  args = {
    ...args,
    label: caption
  }
  return <GlobalButton {...args} />
};

      }
    },
    control: { disable: true }
  }
};


export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'es': return ES.msg.hello;
    case 'fr': return FR.msg.hello;
    case 'kr': return KR.msg.hello;
    case 'zh': return ZH.msg.hello;
    default:
      return EN.msg.hello;
  }
}
